  first_name TEXT NOT NULL,
  last_name TEXT NOT NULL,
  age INTEGER NOT NULL,
  country TEXT NOT NULL,
  phone TEXT NOT NULL,
  balance INTEGER NOT NULL
);

*sql 파일 생성하고 테이블 생성(import 전에 먼저 해야함)
$ aqlite3 mydb.sqlite3

sqlite> .import users.csv users
